	public int OrdenarPorDuracion(Llamada llamada1, Llamada llamada2)
        {
            if(llamada1.Duracion > llamada2.Duracion)
            {
                return -1;
            }
            else if(llamada1.Duracion < llamada2.Duracion)
            {
                return 1;
            }

            return 0;
        }
